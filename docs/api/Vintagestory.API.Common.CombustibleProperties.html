Marks an item as combustible, either by cooking, smelting or firing. This can either imply it is used as a fuel, or can be cooked into another object. Open in GitHub

Examples

Cooking:

"combustiblePropsByType": {
	"bushmeat-raw": {
		"meltingPoint": 150,
		"meltingDuration": 30,
		"smeltedRatio": 1,
		"smeltingType": "cook",
		"smeltedStack": {
			"type": "item",
			"code": "bushmeat-cooked"
		},
		"requiresContainer": false
	}
},

Clay Firing:

"combustiblePropsByType": {
	"bowl-raw": {
		"meltingPoint": 650,
		"meltingDuration": 45,
		"smeltedRatio": 1,
		"smeltingType": "fire",
		"smeltedStack": {
			"type": "block",
			"code": "bowl-fired"
		},
		"requiresContainer": false
	}
},

Fuel Source:

"combustibleProps": {
	"burnTemperature": 1300,
	"burnDuration": 40
},
